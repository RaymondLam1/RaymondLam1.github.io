    Ant Design Pro 集合 Swagger 的坑

    Raymondlam1 2024/1/6

    引入openapi，之后访问localhost:7001/api-docs

            <dependency>
            <groupId>org.springdoc</groupId>
            <artifactId>springdoc-openapi-ui</artifactId>
            <version>1.6.4</version>
        </dependency>

    前端使用ant design pro的配置，参考：https://pro.ant.design/zh-CN/docs/getting-started 。国内建议使用tyarn，使用nvm管理node版本。

    初始化前端工程后，可以删除husky的依赖和配置。

    执行tyarn run openapi，可以在services/目录生成后端相应的接口。有趣的是api-docs刚开始返回的components为{}。这会导致max openapi 执行失败。

    解决的方式是，在后端加入统一的返回接口：

    public class ServiceResponse<T> {
    Boolean success;
    String errorMessage;
    String errorCode;
    T data;

    public static <T> ServiceResponse<T> success() {
        return ServiceResponse.success(null);
    }

    public static <T> ServiceResponse<T> success(T data) {
        return new ServiceResponse<>(true, null, null, data);
    }

    public static <T> ServiceResponse<T> fail(String errorMessage, String errorCode) {
        return new ServiceResponse<>(false, errorMessage, errorCode, null);
    }

    private ServiceResponse() {

    }

    private ServiceResponse(Boolean success, String errorMessage, String errorCode, T data) {
        this.success = success;
        this.errorMessage = errorMessage;
        this.errorCode = errorCode;
        this.data = data;
    }

    public Boolean getSuccess() {
        return success;
    }

    public void setSuccess(Boolean success) {
        this.success = success;
    }

    public String getErrorMessage() {
        return errorMessage;
    }

    public void setErrorMessage(String errorMessage) {
        this.errorMessage = errorMessage;
    }

    public String getErrorCode() {
        return errorCode;
    }

    public void setErrorCode(String errorCode) {
        this.errorCode = errorCode;
    }

    public T getData() {
        return data;
    }

    public void setData(T data) {
        this.data = data;
    }

    @Override
    public String toString() {
        return "ServiceResponse{" +
                "success=" + success +
                ", errorMessage='" + errorMessage + '\'' +
                ", errorCode='" + errorCode + '\'' +
                ", data=" + data +
                '}';
    }
}
